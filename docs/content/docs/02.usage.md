## Loading a design

Pass the `initial-design` prop to pre-load a saved `JSONTemplate`. The editor calls `loadDesign` internally once it's ready, and again whenever the prop value changes.

```vue
<template>
  <EmailEditor :initial-design="myDesign" />
</template>

<script setup lang="ts">
  import type { JSONTemplate } from "#unlayer/props";

  const myDesign = ref<JSONTemplate>(); // loaded from your API
</script>
```

::prose-callout{variant="tip" title="Reactive updates"}
Because `initialDesign` uses a reactive watcher internally, you can swap designs at any time by updating the ref — no need to call `editor.loadDesign()` manually.
::

## Saving a design

Use the `saveDesign` method on the editor instance to get the current design as a `JSONTemplate` object. Store it in your database or `localStorage`.

```vue
<template>
  <EmailEditor @ready="onReady" />
  <button @click="save">Save</button>
</template>

<script setup lang="ts">
  import type { EditorInstance, JSONTemplate } from "#unlayer/props";

  const editor = shallowRef<EditorInstance>();

  const onReady = (instance: EditorInstance | undefined) => {
    editor.value = instance;
  };

  const save = () => {
    editor.value?.saveDesign((design: JSONTemplate) => {
      // persist design to your API
      console.log(design);
    });
  };
</script>
```

## Exporting HTML

Call `exportHtml` to get the final rendered HTML string, ready to send as an email.

```ts
editor.value?.exportHtml((data) => {
  const html = data.html;    // full HTML document
  const json = data.design;  // JSONTemplate (same as saveDesign)
  // send html via your email API
});
```

## Importing a design from a file

Let the user upload a previously exported `.json` design file:

```vue
<template>
  <input ref="fileInput" type="file" accept=".json" hidden @change="onFile" />
  <button @click="fileInput?.click()">Import Design</button>
</template>

<script setup lang="ts">
  import type { EditorInstance } from "#unlayer/props";

  const editor = shallowRef<EditorInstance>();
  const fileInput = useTemplateRef("fileInput");

  const onFile = (e: Event) => {
    const file = (e.target as HTMLInputElement).files?.[0];
    if (!file?.type.includes("json")) return;

    const reader = new FileReader();
    reader.onload = (ev) => {
      const result = ev.target?.result;
      if (typeof result === "string") {
        editor.value?.loadDesign(JSON.parse(result));
      }
    };
    reader.readAsText(file);
  };
</script>
```

## Exporting plain text

Useful for sending a multi-part email with both an HTML and a plain-text version:

```ts
editor.value?.exportPlainText(
  (data) => {
    const text = data.text;
  },
  { ignoreLinks: true, ignoreImages: true },
);
```

## Exporting a PDF or ZIP

Both methods require a connected Unlayer file storage integration.

```ts
// PDF
editor.value?.exportPdf((data) => {
  window.open(data.url!);
});

// ZIP (useful for HTML email templates with assets)
editor.value?.exportZip((data) => {
  window.open(data.url!);
});
```

## Using a stock template

Load one of Unlayer's built-in stock templates by ID:

```ts
editor.value?.loadStockTemplate("your-stock-template-id");
```

Or pass the ID at config time:

```vue
<EmailEditor :stock-template-id="'your-stock-template-id'" />
```
